8  geometry.set_lai(lai_curr + par.dl);
9  auto res_plus = assimilator.net_production(env, &geometry, par, traits);
10  geometry.set_lai(lai_curr);
11 
12  double dnpp_dL = (res_plus.npp - res.npp)/geometry.crown_area/par.dl;
13  double dgpp_dL = (res_plus.gpp - res.gpp)/geometry.crown_area/par.dl;
14  double dE_dL = (res_plus.trans - res.trans)/geometry.crown_area/par.dl;
15 
16  double dL_dt = 0;
17  if (par.optimize_lai) dL_dt = par.response_intensity*(dgpp_dL - par.Chyd*dE_dL - par.Cc*traits.lma);
18  //std::cout << "dnpp_dL = " << dnpp_dL << ", dE_dL = " << 0.001*dE_dL << ", Cc = " << traits.K_leaf << "\n";
19 
20  if (lai_curr < 1) dL_dt = 0; // limit to prevent LAI going negative
21 
22  // calculate and constrain rate of LAI change
23  double max_alloc_lai = par.max_alloc_lai * _dmass_dt_tot; // if npp is negative, there can be no lai increment. if npp is positive, max 10% can be allocated to lai increment
24  bp.dmass_dt_lai = geometry.dmass_dt_lai(dL_dt, max_alloc_lai, traits); // biomass change resulting from LAI change // FIXME: here roots also get shed with LAI. true?
25 
26  return dL_dt;
27 }

141 
142  // NOTE: LAI increment is prioritized (already subtracted from npp above) // if lai is increasing, biomass is partitioned into lai growth and remaining components
143  // if lai is decreasing, lost biomass goes into litter
144  double dmass_dt_nonlai = dm_dt_tot - std::max(dm_dt_lai, 0.0);
145  bp.dmass_dt_lit = std::max(-dm_dt_lai, 0.0);
146 
147  // fraction of biomass going into reproduction and biomass allocation to reproduction
148  double fR = geometry.dreproduction_dmass(par, traits);
149  bp.dmass_dt_rep = fR * dmass_dt_nonlai;
